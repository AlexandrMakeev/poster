package ru.netologi.qa;

public class Radio {

    private int currentNumber;
    private int currentVolume;

    public int getCurrentNumber() {
        return currentNumber;
    }

    public int getCurrentVolume() {
        return currentVolume;
    }


    public int nextNumber() {
        if (currentNumber < 9) {
            currentNumber++;
        } else {
            currentNumber = 0;
        }
        return currentNumber;
    }

    public int prevNumber() {
        if (currentNumber > 0) {
            currentNumber--;
        } else {
            currentNumber = 9;
        }
        return currentNumber;

    }

    public void setCurrentNumber(int newNumber) {
        if (newNumber > 9) {
            return;
        }
        if (newNumber < 0) {
            return;
        }
        currentNumber = newNumber;

    }

    public int pluseVolume() {
        if (currentVolume > 0) {
            currentVolume++;
        } else {
            currentVolume = 0;
        }
        return currentVolume;
    }

    public void prevVolume() {
        if (currentVolume < 10) {
            currentVolume--;
        } else {
            currentVolume = 10;

        }

    }

    public int setCurrentVolume(int currentVolume) {
        if (currentVolume < 0) {
            currentVolume = 0;
        }
        if (currentVolume > 10) {
            currentVolume = 10;
        }
        this.currentVolume = currentVolume;
        return currentVolume;
    }
}
